# loaclStorage еӯҳеӮЁ иҜ»еҸ–

 // е®ҡд№үеӯҳеӮЁзҡ„з©әй—ҙеҗҚ
    let namespace = "mall"
    function saveItem(key, val) { // е…ій”®еӯ—е’ҢеҖј
      // еӯҳеӮЁеүҚпјҢе…ҲиҺ·еҸ–е‘ҪеҗҚз©әй—ҙ
      let storage = window.localStorage.getItem(namespace)
      if (!storage) {
        // еҰӮжһңжІЎжңүе‘ҪеҗҚз©әй—ҙ е°ұеҲқе§ӢеҢ–дёҖдёӘ
        storage = {}
      } else {
        // еҰӮжһңжңүе‘ҪеҗҚз©әй—ҙ е°ұиҪ¬жҲҗеҜ№иұЎж јејҸ
        storage = JSON.parse(storage)
      }
      // иҪ¬жҲҗеҜ№иұЎеҗҺпјҢе°Ҷж•°жҚ®д»Ҙй”®еҖјеҜ№зҡ„ж–№ејҸжҡӮеӯҳ
      storage[key] = val
      // еӯҳеӮЁеҲ°window
      window.localStorage.setItem(namespace, JSON.stringify(storage))
    }
    // йЎөйқўдёӯи°ғз”Ёж–№жі•пјҢеӯҳеӮЁж•°жҚ®
    // saveItem("a", "1")
    function readItem(key, def) {  // иҺ·еҸ–зҡ„е…ій”®еӯ—е’Ңй»ҳи®ӨеҖј
      // иҺ·еҸ– localStorage
      let storage = window.localStorage.getItem(namespace)
      if (!storage) {
        // еҰӮжһңжІЎжңүеӯҳеӮЁ пјҢиҝ”еӣһ й»ҳи®ӨеҖј
        return def
      }
      // еҰӮжһңжңүпјҢиҜ»еҸ–е№¶иҪ¬жҚўдёәеҜ№иұЎ
      storage = JSON.parse(storage)
      let result = storage[key]
      // жңҖеҗҺиҝ”еӣһиҜ»еҸ–еҲ°зҡ„ еҖј жҲ–иҖ… й»ҳи®ӨеҖј
      return result || def
    }
    console.log(readItem("a", "0")); // и°ғз”ЁиҜ»еҸ–зҡ„ж–№жі•

function parseQuery(query) {
    var result = {};

    // еҰӮжһңдёҚжҳҜеӯ—з¬ҰдёІиҝ”еӣһз©әеҜ№иұЎ
    if (typeof query !== 'string') {
        return result;
    }

    // еҺ»жҺүеӯ—з¬ҰдёІејҖеӨҙеҸҜиғҪеёҰзҡ„?
    if (query.charAt(0) === '?') {
        query = query.substring(1);
    }

    var pairs = query.split('&');
    var pair;
    var key, value;
    var i, len;

    for (i = 0, len = pairs.length; i < len; ++i) {
        pair = pairs[i].split('=');
        // application/x-www-form-urlencodedзј–з Ғдјҡе°Ҷ' 'иҪ¬жҚўдёә+
        key = decodeURIComponent(pair[0]).replace(/\+/g, ' ');
        value = decodeURIComponent(pair[1]).replace(/\+/g, ' ');

        // еҰӮжһңжҳҜж–°keyпјҢзӣҙжҺҘж·»еҠ 
        if (!(key in result)) {
            result[key] = value;
        }
        // еҰӮжһңkeyе·Із»ҸеҮәзҺ°дёҖж¬Ўд»ҘдёҠпјҢзӣҙжҺҘеҗ‘ж•°з»„ж·»еҠ value
        else if (isArray(result[key])) {
            result[key].push(value);
        }
        // keyз¬¬дәҢж¬ЎеҮәзҺ°пјҢе°Ҷз»“жһңж”№дёәж•°з»„
        else {
            var arr = [result[key]];
            arr.push(value);
            result[key] = arr;
        } // end if-else
    } // end for

    return result;
}

function parseUrl(url) {
    var result = {};
    var keys = ['href', 'origin', 'protocol', 'host',
                'hostname', 'port', 'pathname', 'search', 'hash'];
    var i, len;
    var regexp = /(([^:]+:)\/\/(([^:\/\?#]+)(:\d+)?))(\/[^?#]*)?(\?[^#]*)?(#.*)?/;

    var match = regexp.exec(url);

    if (match) {
        for (i = keys.length - 1; i >= 0; --i) {
            result[keys[i]] = match[i] ? match[i] : '';
        }
    }

    return result;
}

# FormData з®ҖеҚ•дҪҝз”Ё(ж–№жі•)

ж–Үд»¶дёҠдј formDataдёҠдј д№ӢеүҚжҹҘзңӢеҜ№иұЎйҮҢзҡ„еҖј

  • иҺ·еҸ–дёҖдёӘformиЎЁеҚ•еҜ№иұЎ

let formData = new FormData(form);
  • ж·»еҠ ж•°жҚ® formdata.append(key,value)

formdata.append("b","c");
  • еҲ йҷӨж•°жҚ® formdata.delete(key);

  • и®ҫзҪ®/дҝ®ж”№ж•°жҚ® formdata.set(key,value) еҰӮжһңkeyдёҚеӯҳеңЁеҲҷж–°еўһдёҖжқЎж•°жҚ®пјҢеҰӮжһңеӯҳеңЁпјҢеҲҷдјҡдҝ®ж”№еҜ№еә”зҡ„valueеҖјгҖӮ

  • иҺ·еҸ–(жҹҘзңӢ)ж•°жҚ® formdata.get(key)
var formData = new FormData();
